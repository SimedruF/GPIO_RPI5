1/**
2 * @file gpio_rpi5.h
3 * @brief Public API header for the GPIO_RPI5 library (Raspberry Pi 5).
4 *
5 * Provides a hardware abstraction layer for controlling GPIO pins on the
6 * Raspberry Pi 5 through either direct RP1 register access (mmap) or the
7 * pinctrl CLI backend. The API covers digital I/O, pull resistor
8 * configuration, alternate function selection, and drive strength control.
9 *
10 * Two interchangeable backends implement this API:
11 * - @ref gpio_rpi5.c — Direct register access via `/dev/gpiomem0` (fast, ~10 ns per operation)
12 * - @ref gpio_rpi5_pinctrl.c — `pinctrl` CLI wrapper (slower, ~10 ms per operation, wider GPIO range)
13 *
14 * @note All GPIO numbers follow the RP1/BCM2712 scheme, not physical header pin numbers.
15 * @see https://datasheets.raspberrypi.com/rp1/rp1-peripherals.pdf
16 */

281/**
282 * @brief Open a GPIO pin for use in the specified direction.
283 *
284 * Configures the pin as INPUT or OUTPUT by setting the FUNCSEL to SIO mode
285 * and adjusting the output-enable register. The pin must be opened before
286 * calling pinwrite(), pinread(), or pintoggle().
287 *
288 * @param pin GPIO number (0–53 for mmap backend, 0–237 for pinctrl).
289 * @param mode Direction: INPUT (0) or OUTPUT (1).
290 * @return A pin_t struct with the configured mode. On error, both fields are UNDEF.
291 */
292pin_t pinopen(int pin, int mode);

294/**

339 *

342 * driven — essential for buttons, open-drain buses, etc.

345 * @param pull One of PULL_UP, PULL_DOWN, or PULL_NONE.
