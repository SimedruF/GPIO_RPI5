1/**
2 * @file gpio_rpi5_pinctrl.c
3 * @brief Pinctrl CLI fallback GPIO backend for Raspberry Pi 5.
4 *
5 * This backend executes the `pinctrl` command-line tool (via `popen()`) to
6 * control GPIO pins. It is significantly slower than the mmap backend (~10 ms
7 * per operation due to process spawning) but supports all GPIO ranges:
8 * - RP1 GPIOs 0–53 (40-pin header)
9 * - BCM2712 GPIOs 100–135 (processor-side)
10 * - AON GPIOs 200–237 (always-on domain)
11 *
12 * @par When to Use This Backend
13 * - When you need to access GPIOs outside the RP1 range (100–237).
14 * - When `/dev/gpiomem0` is not available.
15 * - For quick prototyping where speed is not critical.
16 *
17 * @par Limitations
18 * - pin_set_drive() is not supported (returns -1).
19 * - pintoggle() is implemented as pinread() + pinwrite() (two CLI calls).
20 *
21 * Compile with: gcc -c gpio_rpi5_pinctrl.c -o gpio_rpi5_pinctrl.o -Wall -Wextra
22 */

41/**
42 * @brief Validate that a pin index is within any known GPIO range.
43 *
44 * Unlike the mmap backend (which only supports 0–53), this backend accepts:
45 * - 0–53 (RP1 GPIOs)
46 * - 100–135 (BCM2712 GPIOs)
47 * - 200–237 (AON GPIOs)
48 *
49 * @param pin GPIO number to validate.
50 * @return 1 if valid, 0 otherwise.
51 */
52static int gpio_valid_pin(int pin)
53{
54 if (pin >= 0 && pin <= 53) return 1; /* RP1 GPIOs */
55 if (pin >= 100 && pin <= 135) return 1; /* BCM2712 GPIOs */
56 if (pin >= 200 && pin <= 237) return 1; /* AON GPIOs */
57 return 0;
58}

60/**
61 * @brief Initialize the GPIO subsystem (pinctrl backend).
62 *
63 * Resets all pin states to UNDEF and sets the initialization flag.
64 * Unlike the mmap backend, no device files are opened — each
65 * subsequent operation spawns a `pinctrl` CLI process.
66 *
67 * @return Always returns 0 (cannot fail).
68 */
69int gpio_init(void)
70{
71 int i;
72 for (i = 0; i <= GPIO_MAX_INDEX; i++) {
73 rpi5_gpio[i].mode = UNDEF;
74 rpi5_gpio[i].state = UNDEF;
75 }
76 gpio_initialized = 1;
77 return 0;
78}

91/**
92 * @brief Quick pin test: pulse a pin HIGH for 1 second (pinctrl backend).
93 *
94 * Opens the pin as OUTPUT, drives HIGH, waits 1 second, drives LOW,
95 * then closes. Executes 4 `pinctrl` commands total.
96 *
97 * @param pin_indx GPIO number to test (0–53, 100–135, or 200–237).
98 * @return 0 on success, -1 if the pin is invalid or a command fails.
99 */
100int gpio_pintest(int pin_indx)
101{
102 if (!gpio_valid_pin(pin_indx)) return -1;
103
104 pin_t testpin = pinopen(pin_indx, OUTPUT);
105 if (testpin.mode == UNDEF && testpin.state == UNDEF)
106 return -1;
107
108 pinwrite(pin_indx, HIGH);
109 sleep(1);
110 pinwrite(pin_indx, LOW);
111 pinclose(pin_indx);
112 return 0;
113}

115/**
116 * @brief Open a GPIO pin in the specified direction (pinctrl backend).
117 *
118 * Runs `pinctrl set <pin> op` (output) or `pinctrl set <pin> ip` (input)
119 * via popen(). The command spawns a separate process, making this ~1000x
120 * slower than the mmap backend.
121 *
122 * @param pin GPIO number (0–53, 100–135, or 200–237).
123 * @param mode Direction: INPUT (0) or OUTPUT (1).
124 * @return pin_t with the configured mode. On error, both fields are UNDEF.
125 */
126pin_t pinopen(int pin, int mode)
127{
128 char cmd[128];
129 FILE *pf;
130 pin_t err_pin = { UNDEF, UNDEF };
131
132 if (!gpio_valid_pin(pin))
133 {
134 fprintf(stderr, "pinopen: Invalid pin %d\n", pin);
135 return err_pin;
136 }
137
138 if (mode == OUTPUT)
139 snprintf(cmd, sizeof(cmd), "pinctrl set %d op", pin);
140 else
141 snprintf(cmd, sizeof(cmd), "pinctrl set %d ip", pin);
142
143 pf = popen(cmd, "r");
144 if (!pf)
145 {
146 fprintf(stderr, "pinopen: Could not open pipe for pin %d\n", pin);
147 return err_pin;
148 }
149
150 if (pclose(pf) != 0)
151 {
152 fprintf(stderr, "pinopen: Failed to close command stream for pin %d\n", pin);
153 return err_pin;
154 }
155
156 rpi5_gpio[pin].mode = mode;
157 return (pin_t) { UNDEF, mode };
158}

160/**
161 * @brief Close a GPIO pin and reset to input (pinctrl backend).
162 *
163 * Runs `pinctrl set <pin> ip` to reset the pin to a safe input state.
164 * The cached mode and state are reset to UNDEF.
165 *
166 * @param indx_pin GPIO number to close.
167 */
168void pinclose(int indx_pin)
169{
170 char cmd[128];
171 FILE *pf;
172
173 if (!gpio_valid_pin(indx_pin)) return;
174
175 /* Reset pin to input (safe default) */
176 snprintf(cmd, sizeof(cmd), "pinctrl set %d ip", indx_pin);
177 pf = popen(cmd, "r");
178 if (pf) pclose(pf);
179
180 rpi5_gpio[indx_pin].mode = UNDEF;
181 rpi5_gpio[indx_pin].state = UNDEF;
182}

227/**
228 * @brief Read the current logic level of a GPIO pin (pinctrl backend).
229 *
230 * Runs `pinctrl get <pin>` and parses the output for "| lo" or "| hi"
231 * to determine the logic level. The pipe delimiter avoids false matches
232 * with pin name substrings.
233 *
234 * @param pin GPIO number.
235 * @return HIGH (1), LOW (0), or -1 on error or if the level cannot be parsed.
236 */
237int pinread(int pin)
238{
239 char cmd[128];
240 FILE *pf;
241 char data[512];
242 int pinout = -1;
243
244 if (!gpio_valid_pin(pin))
245 {
246 fprintf(stderr, "pinread: Invalid pin %d\n", pin);
247 return -1;
248 }
249
250 snprintf(cmd, sizeof(cmd), "pinctrl get %d", pin);
251
252 pf = popen(cmd, "r");
253 if (!pf)
254 {
255 fprintf(stderr, "pinread: Could not open pipe for pin %d\n", pin);
256 return -1;
257 }
258
259 if (fgets(data, sizeof(data), pf) != NULL)
260 {
261 /* Search for "| lo" or "| hi" to avoid matching pin name substrings */
262 if (strstr(data, "| lo"))
263 pinout = 0;
264 else if (strstr(data, "| hi"))
265 pinout = 1;
266 }
267
268 if (pclose(pf) != 0)
269 {
270 fprintf(stderr, "pinread: Failed to close command stream for pin %d\n", pin);
271 return -1;
272 }
273
274 if (pinout >= 0)
275 rpi5_gpio[pin].state = pinout;
276
277 return pinout;
278}

280/**
281 * @brief Configure the internal pull resistor (pinctrl backend).
282 *
283 * Runs `pinctrl set <pin> pu|pd|pn` to set pull-up, pull-down,
284 * or no-pull respectively.
285 *
286 * @param pin GPIO number.
287 * @param pull Pull mode: PULL_UP (1), PULL_DOWN (2), or PULL_NONE (0).
288 * @return 0 on success, -1 on error.
289 */
290int pinpull(int pin, int pull)
291{
292 char cmd[128];
293 FILE *pf;
294 const char *pull_str;
295
296 if (!gpio_valid_pin(pin))
297 {
298 fprintf(stderr, "pinpull: Invalid pin %d\n", pin);
299 return -1;
300 }
301
302 switch (pull)
303 {
304 case PULL_UP: pull_str = "pu"; break;
305 case PULL_DOWN: pull_str = "pd"; break;
306 case PULL_NONE: pull_str = "pn"; break;
307 default:
308 fprintf(stderr, "pinpull: Invalid pull mode %d\n", pull);
309 return -1;
310 }
311
312 snprintf(cmd, sizeof(cmd), "pinctrl set %d %s", pin, pull_str);
313
314 pf = popen(cmd, "r");
315 if (!pf)
316 {
317 fprintf(stderr, "pinpull: Could not open pipe for pin %d\n", pin);
318 return -1;
319 }
320
321 if (pclose(pf) != 0)
322 {
323 fprintf(stderr, "pinpull: Failed to close command stream for pin %d\n", pin);
324 return -1;
325 }
326
327 return 0;
328}

330/**
331 * @brief Toggle the output state of a GPIO pin (pinctrl backend).
332 *
333 * Reads the current level via pinread(), then writes the inverted value
334 * via pinwrite(). Requires two `pinctrl` commands (slower than mmap XOR).
335 *
336 * @param pin GPIO number (should have been opened as OUTPUT).
337 * @return 0 on success, -1 on error.
338 */
339int pintoggle(int pin)
340{
341 int val = pinread(pin);
342 if (val < 0) return -1;
343 return pinwrite(pin, val ? LOW : HIGH);
344}
