1/**
2 * @file gpio_rpi5.c
3 * @brief Direct register access GPIO backend for Raspberry Pi 5.
4 *
5 * This backend uses memory-mapped I/O (`mmap`) on `/dev/gpiomem0` to access
6 * the RP1 southbridge GPIO registers directly from userspace. It provides
7 * extremely fast GPIO operations (~10 ns per read/write) compared to the
8 * pinctrl CLI backend (~10 ms).
9 *
10 * @par Supported GPIO Range
11 * Only RP1 GPIOs 0–53 are accessible through this backend. For BCM2712
12 * GPIOs (100–135) and AON GPIOs (200–237), use the pinctrl backend instead.
13 *
14 * @par RP1 Memory Map (via /dev/gpiomem0, 192 KB region)
15 * | Block | Offset | Description |
16 * |--------------|----------|--------------------------------------|
17 * | IO_BANK0 | 0x00000 | GPIO 0–27 control/status registers |
18 * | IO_BANK1 | 0x04000 | GPIO 28–53 control/status registers |
19 * | SYS_RIO0 | 0x10000 | GPIO 0–27 fast register I/O |
20 * | SYS_RIO1 | 0x14000 | GPIO 28–53 fast register I/O |
21 * | PADS_BANK0 | 0x20000 | GPIO 0–27 pad configuration |
22 * | PADS_BANK1 | 0x24000 | GPIO 28–53 pad configuration |
23 *
24 * @par Atomic Register Access
25 * The SYS_RIO blocks support atomic SET, CLR, and XOR operations at fixed
26 * offsets from the base, allowing glitch-free single-register writes without
27 * read-modify-write races.
28 *
29 * @see https://datasheets.raspberrypi.com/rp1/rp1-peripherals.pdf
30 *
31 * Compile with: gcc -c gpio_rpi5.c -o gpio_rpi5.o -Wall -Wextra
32 */

61/** @brief Device path for the RP1 GPIO memory region. */
62#define GPIOMEM_DEVICE "/dev/gpiomem0"
63
64/** @brief Total size of the memory-mapped region (192 KB). */
65#define GPIO_MEM_SIZE 0x30000
66
67/** @name IO Bank Base Offsets
68 * Control and status registers for each GPIO pin.
69 * @{ */
70#define RP1_IO_BANK0 0x00000 /**< @brief GPIO 0–27 control registers. */
71#define RP1_IO_BANK1 0x04000 /**< @brief GPIO 28–53 control registers. */
72/** @} */
73
74/** @name RIO (Register I/O) Base Offsets
75 * Fast read/write/toggle registers with atomic access support.
76 * @{ */
77#define RP1_SYS_RIO0 0x10000 /**< @brief GPIO 0–27 fast register I/O. */
78#define RP1_SYS_RIO1 0x14000 /**< @brief GPIO 28–53 fast register I/O. */
79/** @} */
80
81/** @name Pad Control Base Offsets
82 * Drive strength, pull resistor, slew rate, and Schmitt trigger configuration.
83 * @{ */
84#define RP1_PADS_BANK0 0x20000 /**< @brief GPIO 0–27 pad configuration. */
85#define RP1_PADS_BANK1 0x24000 /**< @brief GPIO 28–53 pad configuration. */
86/** @} */
87
88/** @name Atomic Access Offsets
89 * Added to the RIO base to perform atomic bit operations without read-modify-write.
90 * @{ */
91#define RP1_XOR_OFFSET 0x1000 /**< @brief XOR (toggle) bits in the register. */
92#define RP1_SET_OFFSET 0x2000 /**< @brief SET (write 1) bits in the register. */
93#define RP1_CLR_OFFSET 0x3000 /**< @brief CLR (write 0) bits in the register. */
94/** @} */
95
96/** @name RIO Register Offsets
97 * Offsets within each SYS_RIOx block for output, output-enable, and input.
98 * @{ */
99#define RIO_OUT 0x00 /**< @brief Output value register. */
100#define RIO_OE 0x04 /**< @brief Output enable register (1 = output, 0 = input). */
101#define RIO_IN 0x08 /**< @brief Input value register (read-only). */
102/** @} */
103
104/** @name IO Bank Per-Pin Register Macros
105 * Each pin occupies 8 bytes: STATUS (4 bytes) + CTRL (4 bytes).
106 * @{ */
107#define GPIO_STATUS(pin_in_bank) ((pin_in_bank) * 8) /**< @brief STATUS register offset for a pin within its bank. */
108#define GPIO_CTRL(pin_in_bank) ((pin_in_bank) * 8 + 4) /**< @brief CTRL register offset for a pin within its bank. */
109/** @} */
110
111/** @name GPIO CTRL Register Fields
112 * @{ */
113#define CTRL_FUNCSEL_MASK 0x1F /**< @brief FUNCSEL field mask (bits 4:0). */
114#define CTRL_FUNCSEL_LSB 0 /**< @brief FUNCSEL field LSB position. */
115#define CTRL_OUTOVER_MASK (0x3 << 12) /**< @brief Output override field mask. */
116#define CTRL_OEOVER_MASK (0x3 << 14) /**< @brief Output-enable override field mask. */
117/** @} */
118
119/** @name PADS Register Macros and Bit Fields
120 * @{ */
121#define PADS_GPIO(pin_in_bank) (0x04 + (pin_in_bank) * 4) /**< @brief Pad register offset (pin 0 starts at 0x04). */
122#define PAD_OD_BIT (1 << 7) /**< @brief Output disable bit. */
123#define PAD_IE_BIT (1 << 6) /**< @brief Input enable bit. */
124#define PAD_DRIVE_MASK (0x3 << 4) /**< @brief Drive strength field mask. */
125#define PAD_DRIVE_LSB 4 /**< @brief Drive strength field LSB position. */
126#define PAD_PUE_BIT (1 << 3) /**< @brief Pull-up enable bit. */
127#define PAD_PDE_BIT (1 << 2) /**< @brief Pull-down enable bit. */
128#define PAD_SCHMITT_BIT (1 << 1) /**< @brief Schmitt trigger enable bit. */
129#define PAD_SLEWFAST_BIT (1 << 0) /**< @brief Fast slew rate enable bit. */
130/** @} */
131

170/**
171 * @brief Resolve a GPIO number (0–53) into its bank-specific parameters.
172 *
173 * The RP1 has two banks: Bank 0 covers GPIO 0–27, Bank 1 covers GPIO 28–53.
174 * This function returns the base offsets for IO, RIO, and PADS registers,
175 * along with the bit position within the bank.
176 *
177 * @param[in] pin GPIO number (0–53).
178 * @param[out] io_base IO bank base offset.
179 * @param[out] rio_base RIO bank base offset.
180 * @param[out] pads_base PADS bank base offset.
181 * @param[out] bit Bit position of the pin within the bank (0–27).
182 * @return 0 on success, -1 if the pin is out of the RP1 range.
183 */
184static int pin_to_bank(int pin,
185 uint32_t *io_base,
186 uint32_t *rio_base,
187 uint32_t *pads_base,
188 int *bit)
189{
190 if (pin >= 0 && pin <= 27)
191 {
192 *io_base = RP1_IO_BANK0;
193 *rio_base = RP1_SYS_RIO0;
194 *pads_base = RP1_PADS_BANK0;
195 *bit = pin;
196 return 0;
197 }
198 if (pin >= 28 && pin <= 53)
199 {
200 *io_base = RP1_IO_BANK1;
201 *rio_base = RP1_SYS_RIO1;
202 *pads_base = RP1_PADS_BANK1;
203 *bit = pin - 28;
204 return 0;
205 }
206 return -1;
207}

221/**
222 * @brief Initialize the GPIO subsystem (mmap backend).
223 *
224 * Opens `/dev/gpiomem0` and maps 192 KB of RP1 register space into the
225 * process address space. All pin states are reset to UNDEF.
226 *
227 * Must be called exactly once before any other GPIO function.
228 *
229 * @return 0 on success, -1 if the device cannot be opened or mmap fails.
230 * @note Requires read/write access to `/dev/gpiomem0` (default on Raspberry Pi OS).
231 */
232int gpio_init(void)
233{
234 int i;
235
236 gpio_mmap_fd = open(GPIOMEM_DEVICE, O_RDWR | O_SYNC);
237 if (gpio_mmap_fd < 0)
238 {
239 perror("gpio_init: Failed to open " GPIOMEM_DEVICE);
240 return -1;
241 }
242
243 gpio_mmap_base = (volatile uint32_t *)mmap(
244 NULL,
245 GPIO_MEM_SIZE,
246 PROT_READ | PROT_WRITE,
247 MAP_SHARED,
248 gpio_mmap_fd,
249 0);
250
251 if (gpio_mmap_base == MAP_FAILED)
252 {
253 perror("gpio_init: mmap failed");
254 close(gpio_mmap_fd);
255 gpio_mmap_fd = -1;
256 gpio_mmap_base = NULL;
257 return -1;
258 }
259
260 for (i = 0; i <= GPIO_MAX_INDEX; i++)
261 {
262 rpi5_gpio[i].mode = UNDEF;
263 rpi5_gpio[i].state = UNDEF;
264 }
265
266 gpio_initialized = 1;
267 return 0;
268}

270/**
271 * @brief Release all GPIO resources (mmap backend).
272 *
273 * Unmaps the RP1 register region and closes the `/dev/gpiomem0` file
274 * descriptor. Safe to call even if gpio_init() was not called or already
275 * cleaned up.
276 */
277void gpio_cleanup(void)
278{
279 if (gpio_mmap_base && gpio_mmap_base != MAP_FAILED)
280 {
281 munmap((void *)gpio_mmap_base, GPIO_MEM_SIZE);
282 gpio_mmap_base = NULL;
283 }
284 if (gpio_mmap_fd >= 0)
285 {
286 close(gpio_mmap_fd);
287 gpio_mmap_fd = -1;
288 }
289 gpio_initialized = 0;
290}

292/**
293 * @brief Open a GPIO pin in the specified direction (mmap backend).
294 *
295 * Configures the pin for standard GPIO (SIO) operation by:
296 * 1. Setting FUNCSEL to SIO (ALT5) in the IO_BANK CTRL register.
297 * 2. Clearing output/OE overrides so SIO controls the pin.
298 * 3. Enabling input (IE) and clearing output-disable (OD) in the pad register.
299 * 4. Setting or clearing the Output Enable bit via atomic RIO registers.
300 *
301 * @param pin GPIO number (0–53).
302 * @param mode Direction: INPUT (0) or OUTPUT (1).
303 * @return pin_t with the configured mode. On error, both fields are UNDEF.
304 */
305pin_t pinopen(int pin, int mode)
306{
307 uint32_t io_base, rio_base, pads_base;
308 int bit;
309 uint32_t ctrl_val, pad_val;
310 pin_t err_pin = { UNDEF, UNDEF };
311
312 if (!gpio_valid_pin(pin))
313 {
314 fprintf(stderr, "pinopen: Invalid pin %d (only 0-53 supported)\n", pin);
315 return err_pin;
316 }
317 if (!gpio_initialized)
318 {
319 fprintf(stderr, "pinopen: gpio_init() not called\n");
320 return err_pin;
321 }
322 if (pin_to_bank(pin, &io_base, &rio_base, &pads_base, &bit) < 0)
323 return err_pin;
324
325 /* Set FUNCSEL to SIO (5) for normal GPIO operation */
326 ctrl_val = reg_read(io_base + GPIO_CTRL(bit));
327 ctrl_val &= ~CTRL_FUNCSEL_MASK;
328 ctrl_val |= GPIO_FUNC_SIO;
329 /* Clear overrides: let SIO control output/OE */
330 ctrl_val &= ~(CTRL_OUTOVER_MASK | CTRL_OEOVER_MASK);
331 reg_write(io_base + GPIO_CTRL(bit), ctrl_val);
332
333 /* Configure pad: enable input, clear output-disable */
334 pad_val = reg_read(pads_base + PADS_GPIO(bit));
335 pad_val |= PAD_IE_BIT; /* Input enable (needed for readback) */
336 pad_val &= ~PAD_OD_BIT; /* Clear output disable */
337 reg_write(pads_base + PADS_GPIO(bit), pad_val);
338
339 if (mode == OUTPUT)
340 {
341 /* Set Output Enable bit via atomic SET register */
342 reg_write(rio_base + RP1_SET_OFFSET + RIO_OE, (1u << bit));
343 }
344 else
345 {
346 /* Clear Output Enable bit via atomic CLR register */
347 reg_write(rio_base + RP1_CLR_OFFSET + RIO_OE, (1u << bit));
348 }
349
350 rpi5_gpio[pin].mode = mode;
351 return (pin_t) { UNDEF, mode };
352}

354/**
355 * @brief Close a GPIO pin and reset it to a safe state (mmap backend).
356 *
357 * Clears the Output Enable bit (sets pin to input) and sets FUNCSEL to
358 * GPIO_FUNC_NULL (31) to disconnect the pin from all peripherals.
359 * The cached state is reset to UNDEF.
360 *
361 * @param indx_pin GPIO number (0–53) to close.
362 */
363void pinclose(int indx_pin)
364{
365 uint32_t io_base, rio_base, pads_base;
366 int bit;
367 uint32_t ctrl_val;
368
369 if (!gpio_valid_pin(indx_pin) || !gpio_initialized) return;
370 if (pin_to_bank(indx_pin, &io_base, &rio_base, &pads_base, &bit) < 0) return;
371
372 /* Reset to input (safe default) */
373 reg_write(rio_base + RP1_CLR_OFFSET + RIO_OE, (1u << bit));
374
375 /* Set FUNCSEL to NULL (disable) */
376 ctrl_val = reg_read(io_base + GPIO_CTRL(bit));
377 ctrl_val &= ~CTRL_FUNCSEL_MASK;
378 ctrl_val |= GPIO_FUNC_NULL;
379 reg_write(io_base + GPIO_CTRL(bit), ctrl_val);
380
381 rpi5_gpio[indx_pin].mode = UNDEF;
382 rpi5_gpio[indx_pin].state = UNDEF;
383}

385/**
386 * @brief Write a digital value to a GPIO output pin (mmap backend).
387 *
388 * Uses the atomic SET or CLR register on SYS_RIO to change the output
389 * level in a single write, avoiding read-modify-write glitches.
390 *
391 * @param indx_pin GPIO number (0–53, must have been opened as OUTPUT).
392 * @param value Desired logic level: HIGH (1) or LOW (0).
393 * @return 0 on success, -1 on error (invalid pin or not initialized).
394 */
395int pinwrite(int indx_pin, int value)
396{
397 uint32_t io_base, rio_base, pads_base;
398 int bit;
399
400 if (!gpio_valid_pin(indx_pin))
401 {
402 fprintf(stderr, "pinwrite: Invalid pin %d\n", indx_pin);
403 return -1;
404 }
405 if (!gpio_initialized)
406 {
407 fprintf(stderr, "pinwrite: gpio_init() not called\n");

422/**
423 * @brief Read the current logic level of a GPIO pin (mmap backend).
424 *
425 * Reads the RIO_IN input register and extracts the bit for the given pin.
426 * Works on both INPUT and OUTPUT pins (output readback is supported).
427 *
428 * @param pin GPIO number (0–53).
429 * @return HIGH (1), LOW (0), or -1 on error.
430 */
431int pinread(int pin)
432{
433 uint32_t io_base, rio_base, pads_base;
434 int bit;
435 uint32_t val;
436 int result;
437
438 if (!gpio_valid_pin(pin))
439 {
440 fprintf(stderr, "pinread: Invalid pin %d\n", pin);
441 return -1;
442 }
443 if (!gpio_initialized)
444 {

464 * @param pin GPIO number (0–53, must have been opened as OUTPUT).
465 * @return 0 on success, -1 on error.

467int pintoggle(int pin)

470 int bit;
471
472 if (!gpio_valid_pin(pin))
473 {

493/**
494 * @brief Configure the internal pull resistor for a GPIO pin (mmap backend).
495 *
496 * Modifies the PUE (pull-up enable) and PDE (pull-down enable) bits in the
497 * pad register. Both bits are cleared first, then the requested pull is set.
498 *
499 * @param pin GPIO number (0–53).
500 * @param pull Pull mode: PULL_UP (1), PULL_DOWN (2), or PULL_NONE (0).
501 * @return 0 on success, -1 on error (invalid pin, not initialized, or bad pull value).
502 */
503int pinpull(int pin, int pull)
504{
505 uint32_t io_base, rio_base, pads_base;
506 int bit;
507 uint32_t pad_val;
508
509 if (!gpio_valid_pin(pin))
510 {
511 fprintf(stderr, "pinpull: Invalid pin %d\n", pin);
512 return -1;
513 }
514 if (!gpio_initialized)
515 {
516 fprintf(stderr, "pinpull: gpio_init() not called\n");
517 return -1;
518 }
519 if (pin_to_bank(pin, &io_base, &rio_base, &pads_base, &bit) < 0)
520 return -1;
521
522 pad_val = reg_read(pads_base + PADS_GPIO(bit));
523
524 /* Clear both pull bits first */
525 pad_val &= ~(PAD_PUE_BIT | PAD_PDE_BIT);
526
527 switch (pull)
528 {
529 case PULL_UP:
530 pad_val |= PAD_PUE_BIT;
531 break;
532 case PULL_DOWN:
533 pad_val |= PAD_PDE_BIT;
534 break;
535 case PULL_NONE:
536 /* Both bits already cleared */
537 break;
538 default:
539 fprintf(stderr, "pinpull: Invalid pull mode %d\n", pull);
540 return -1;
541 }
542
543 reg_write(pads_base + PADS_GPIO(bit), pad_val);
544 return 0;
545}

547/**
548 * @brief Set the alternate function (FUNCSEL) for a GPIO pin (mmap backend).
549 *
550 * Switches the pin from standard GPIO mode to a hardware peripheral function
551 * by writing the function number into the CTRL register's FUNCSEL field.
552 * Also enables input and clears output-disable on the pad.
553 *
554 * @param pin GPIO number (0–53).
555 * @param func Alternate function: ALT0–ALT8 (0–8) or GPIO_FUNC_NULL (31).
556 * @return 0 on success, -1 on error.
557 *
558 * @par Example
559 * @code
560 * pinalt(GPIO2, ALT3); // Set GPIO2 to I2C1 SDA
561 * pinalt(GPIO3, ALT3); // Set GPIO3 to I2C1 SCL
562 * @endcode
563 */
564int pinalt(int pin, int func)
565{
566 uint32_t io_base, rio_base, pads_base;
567 int bit;
568 uint32_t ctrl_val, pad_val;
569
570 if (!gpio_valid_pin(pin))
571 {
572 fprintf(stderr, "pinalt: Invalid pin %d\n", pin);
573 return -1;
574 }
575 if (!gpio_initialized)
576 {
577 fprintf(stderr, "pinalt: gpio_init() not called\n");
578 return -1;
579 }
580 if (func < 0 || (func > 8 && func != GPIO_FUNC_NULL))
581 {
582 fprintf(stderr, "pinalt: Invalid function %d for pin %d\n", func, pin);
583 return -1;
584 }
585 if (pin_to_bank(pin, &io_base, &rio_base, &pads_base, &bit) < 0)
586 return -1;
587
588 /* Set FUNCSEL in CTRL register */
589 ctrl_val = reg_read(io_base + GPIO_CTRL(bit));
590 ctrl_val &= ~CTRL_FUNCSEL_MASK;
591 ctrl_val |= (func & CTRL_FUNCSEL_MASK);
592 reg_write(io_base + GPIO_CTRL(bit), ctrl_val);
593
594 /* Ensure pad input is enabled for the peripheral */
595 pad_val = reg_read(pads_base + PADS_GPIO(bit));
596 pad_val |= PAD_IE_BIT;
597 pad_val &= ~PAD_OD_BIT;
598 reg_write(pads_base + PADS_GPIO(bit), pad_val);
599
600 return 0;
601}

603/**
604 * @brief Set the output drive strength for a GPIO pin (mmap backend).
605 *
606 * Modifies the DRIVE field (bits 5:4) in the pad register to control
607 * the maximum source/sink current. Higher drive is useful for fast signals
608 * (SPI clock) or pins driving long traces.
609 *
610 * @param pin GPIO number (0–53).
611 * @param strength One of: DRIVE_2MA (0), DRIVE_4MA (1), DRIVE_8MA (2), DRIVE_12MA (3).
612 * @return 0 on success, -1 on error.
613 */
614int pin_set_drive(int pin, int strength)
615{
616 uint32_t io_base, rio_base, pads_base;
617 int bit;
618 uint32_t pad_val;
619
620 if (!gpio_valid_pin(pin))
621 {
622 fprintf(stderr, "pin_set_drive: Invalid pin %d\n", pin);
623 return -1;
624 }
625 if (!gpio_initialized)
626 {
627 fprintf(stderr, "pin_set_drive: gpio_init() not called\n");
628 return -1;
629 }
630 if (strength < DRIVE_2MA || strength > DRIVE_12MA)
631 {
632 fprintf(stderr, "pin_set_drive: Invalid strength %d\n", strength);
633 return -1;
634 }
635 if (pin_to_bank(pin, &io_base, &rio_base, &pads_base, &bit) < 0)
636 return -1;
637
638 pad_val = reg_read(pads_base + PADS_GPIO(bit));
639 pad_val &= ~PAD_DRIVE_MASK;
640 pad_val |= ((strength << PAD_DRIVE_LSB) & PAD_DRIVE_MASK);
641 reg_write(pads_base + PADS_GPIO(bit), pad_val);
642
643 return 0;
644}

647 * @brief Quick pin test: pulse a pin HIGH for 1 second, then LOW (mmap backend).
648 *
649 * Opens the pin as OUTPUT, drives it HIGH, waits 1 second, drives LOW,
650 * then closes the pin. Useful for visual verification with an LED or
651 * logic analyzer.
652 *
653 * @param pin_indx GPIO number (0–53) to test.
654 * @return 0 on success, -1 if the pin is invalid or cannot be opened.
655 */
656int gpio_pintest(int pin_indx)
657{
658 if (!gpio_valid_pin(pin_indx)) return -1;
659
660 pin_t testpin = pinopen(pin_indx, OUTPUT);
661 if (testpin.mode == UNDEF && testpin.state == UNDEF)
662 return -1;
663
664 pinwrite(pin_indx, HIGH);
665 sleep(1);
666 pinwrite(pin_indx, LOW);
667 pinclose(pin_indx);
668 return 0;
669}
